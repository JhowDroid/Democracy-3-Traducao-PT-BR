[dilemma]
name=REWARDSFORFAILURE
guiname=Limitar Bonus Bancários
description=Houve apelos na mídia para o governo fazer algo sobre os grandes pagamentos de bônus que estão sendo feitas para CEOs da empresa, apesar de queda dos lucros. Nos casos em que as empresas não conseguiram ter lucro, o governo tem sido solicitado para forçar as empresas a reter os pagamentos de bônus de executivos de topo.

[influences]
0 = _default_,0.9+(0*x)
1 = GDP,0-(0.4*x)

[option0]
Name = Restringir bônus
Description = Por muito tempo, a cultura de bônus entre CEOs ficou para correr fora de controle. Os bônus são suposto ser uma recompensa para o sucesso do negócio, mas executivos de bancos de topo estão tomando bônus de não ganhos, enquanto cortam o número de funcionários. Precisamos de uma lei que impede as empresas de pagar um bônus em anos onde eles não conseguiram ter um lucro.
OnImplement = CreateGrudge(Curb Banking Bonuses,REWARDSFORFAILURE,Capitalist,-0.08,0.9f);CreateGrudge(Curb Banking Bonuses,REWARDSFORFAILURE,TradeUnionist,0.08,0.9f);CreateGrudge(Curb Banking Bonuses,REWARDSFORFAILURE,Socialist,0.08,0.9f);CreateGrudge(Curb Banking Bonuses,REWARDSFORFAILURE,Wealthy,-0.04,0.9f);CreateGrudge(Curb Banking Bonuses,REWARDSFORFAILURE,BrainDrain,0.073,0.9f);

[option1]
Name = Fazer nada
Description = É fácil ser crítico de pessoas que ganham grandes salários e bônus, mas o fato é que vivemos em um mundo globalizado e bônus superiores são necessários para atrair os melhores talentos. Isto é interferência desnecessária do governo na estrutura de remuneração das empresas e vai machucar a nossa competitividade.
