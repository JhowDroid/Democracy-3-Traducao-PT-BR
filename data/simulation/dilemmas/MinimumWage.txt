[dilemma]
name=MINIMUM WAGE
guiname=Salário mínimo
description=No momento não temos um salário mínimo e há uma proposta de introdução de um limite abaixo do qual seria ilegal a empregar alguém. Isto é projetado para evitar que as pessoas que trabalham para os níveis salariais de trabalho escravo.

[influences]
0 = _default_,0.3+(0*x)
1 = PovertyRate,0+(0.6*x)
2 = _LowIncome,0-(0.5*x)

[option0]
Name = Introduzir lei
Description = Devemos introduzir um salário mínimo imediatamente. Há demasiados casos em que as pequenas empresas estão empregando pessoas para trabalhar na construção ou indústria de alimentos para assustadoramente baixos salários. Ao definir um salário mínimo que podemos garantir a todos os cidadãos que trabalham têm uma boa qualidade de vida.
OnImplement = CreateGrudge(Minimum wage,MINIMUM WAGE,_LowIncome,0.16,1.0);CreateGrudge(Minimum wage,MINIMUM WAGE,PovertyRate,-0.10,1.0);CreateGrudge(Minimum wage,MINIMUM WAGE,Socialist,0.19,1.0);CreateGrudge(Minimum wage,MINIMUM WAGE,Capitalist,-0.08,1.0);CreateGrudge(Minimum wage,MINIMUM WAGE,TradeUnionist,0.22,1.0);CreateGrudge(Minimum wage,MINIMUM WAGE,Wages,0.25,1.0);

[option1]
Name = Rejeitar Lei
Description = Isto é a interferência no mercado livre que é muito pior. Ninguém está forçando as pessoas a trabalhar por estes salários, eles são os salários que o mercado se estabeleceu. Tornando ilegal pagar estes níveis só vai destruir empregos e prejudicar as companhias menores, em particular. O governo não deve interferir no mercado.
