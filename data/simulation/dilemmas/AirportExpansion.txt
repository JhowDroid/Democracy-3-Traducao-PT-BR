name=AirportExpansion
guiname=Expansão do Aeroporto
description=A permissão de planejamento foi requisitada para a expansão de alguns dos maiores aeroportos do nosso país. Simplesmente não há capacidade suficiente para o número de voos que desejam pousar aqui.

[influences]
0 = _random_,0.2,0.4
1 = GDP,0+(0.6*x)

[option0]
Name = Permitir Planos de Expansão
Description = Nossa indústria de turismo é demasiada dependente de viagens aérias, e se expandindo o turismo, proporcionando mais empregos e um crescimento do PIB, nós devemos aumentar a capacidade aérea disponível. Nós precisamos agir nesse momento para garantir o aumento da capacidade em tempo.
OnImplement = CreateGrudge(Airport Expansion,AirportExpansion,Environmentalist,-0.08,1.0);CreateGrudge(Airport Expansion,AirportExpansion,Tourism,0.06,1.0);CreateGrudge(Airport Expansion,AirportExpansion,AirTravel,0.11,1.0);

[option1]
Name = Rejeitar Planos
Description = Viagem aérea é um pesadelo de poluição, tanto o ar e a pluição sonora estão causando problemas, não devemos incentivar a viagem aérea. Além disso, os planos significariam a demolição de inúmeras casas e transtornos as pessoas que vivem na área dos aeroportos. Rejeite os planos!
OnImplement = CreateGrudge(Airport Expansion,AirportExpansion,Environmentalist,0.07,1.0);CreateGrudge(Airport Expansion,AirportExpansion,Tourism,-0.08,1.0);CreateGrudge(Airport Expansion,AirportExpansion,Capitalist,-0.07,1.0);
